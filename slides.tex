\begin{frame}{Aplicar fuerzas}\usebeamercolor[fg]{normal text}
  \begin{block}{Objetivos}
  \begin{itemize}
    \item Actualizar las velocidades de los cuerpos.
    \item En base a:
    \begin{itemize}
      \item Duración del ciclo de simulación.
      \item Fuerzas externas.
      \item Velocidad anterior.
    \end{itemize}
  \end{itemize}
  \end{block}
  \begin{block}{Cálculo (método de Euler y 2ª ley de Newton)}
  \begin{columns}[t]
    \column{0.6\textwidth}{
      \begin{align*} 
        v(t + \Delta t) &= v(t) + \Delta t \cdot m^{-1} \cdot f(t + \Delta t) \\
        \omega(t + \Delta t) &= \omega(t) + \Delta t \cdot I^{-1} \cdot \tau(t + \Delta t)
      \end{align*}
    }
    \column{0.4\textwidth}{
      \begin{itemize}
        \item[$\Delta t$]: Duración ciclo.
        \item[$f$]: Fuerza.
        \item[$\tau$]: Torque.
      \end{itemize}
    }
  \end{columns}
  \end{block}
\end{frame}

\begin{frame}{Resolver restricciones: Colisiones}\usebeamercolor[fg]{normal text}
  \begin{columns}
    \column{0.75\textwidth}{
    \begin{block}{Objetivos}
    \begin{itemize}
      \item Por cada árbitro de colisión:
      \begin{itemize}
        \item Calcular el impulso para evitar la penetración.
        \item Aplicar su fuerza normal y tangencial.
      \end{itemize}
    \end{itemize}
    \end{block}
    \begin{block}{Calculo (3ª ley de Newton)}
    \begin{itemize}
      \item Velocidad relativa al punto de contacto:
      \begin{align*}
        \Delta v &= (v_2 + \omega_2 \times r_2) - (v_1 + \omega_1 \times r_1)
      \end{align*}
      \item Velocidad sobre el vector normal:
      \begin{align*}
        v_n &= \Delta v \cdot n
      \end{align*}
      \item Restricción de no-penetración:
      \begin{align*}
        v_n >= 0
      \end{align*}
    \end{itemize}
    \end{block}
  }
  \column{0.25\textwidth}{

\end{frame}

\subsection{Estructuras de aceleración: Índices espaciales}

\begin{frame}{¿Qué es un índice espacial?}\usebeamercolor[fg]{normal text}
  \begin{block}{Definición}
  \begin{itemize}
    \item Estructura de aceleración para consultas espaciales.
    \item Filtrar las geometrías en un espacio \emph{2D} o 3D por área.
  \end{itemize}
  \end{block}
  \begin{block}{Aplicaciones}
  \begin{itemize}
    \item Sistemas de Información Geográfica.
    \item Computación gráfica (\textit{Ray Tracing}).
    \item \emph{Motores de físicas} y gráficos.
  \end{itemize}
  \end{block}
  \begin{block}{Tipos}
  \begin{itemize}
    \item Estáticos (geometrías quietas):
    \begin{itemize}
      \item \textit{Octrees/Quadtrees}, \textit{Bounding Volume Hierarchy}, \textit{Kd-Trees}, \ldots
    \end{itemize}
    \item Dinámicos (geometrías en movimiento):
    \begin{itemize}
      \item \textit{\textbf{Dynamic Bounding Volume Hierarchy}}, \textit{\textbf{Sweep and Prune}}, \ldots
    \end{itemize}
  \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{\textit{Sweep and Prune}\usebeamercolor[fg]{normal text}}
  \begin{quote}
    Especializado para la \textit{broad-phase}.
  \end{quote}

  \begin{block}{Algoritmo}
  \begin{itemize}
    \item Guardar la coordenada mínima y máxima de cada cuerpo en cada eje.
    \item Ordenar la lista de coordenadas de cada eje.
    \item Detectar los pares de geometrías solapantes.
    \begin{itemize}
      \item Coordenada de un cuerpo dentro de la mínima y máxima de otro.
      \item Solapamiento si pasa en todos los ejes.
    \end{itemize}
  \end{itemize}
  \end{block}
\end{frame}

\section{Objetivos y motivaciones}

\begin{frame}{Mejorar el rendimiento de Box2D-lite}\usebeamercolor[fg]{normal text}

  \begin{block}{¿Por qué?}
  \begin{itemize}
    \item Buena documentación.
    \item Código fuente limpio y comprensible.
    \item Alto coste computacional.
    \item Carente de muchas optimizaciones.
    \item No aprovecha las arquitecturas multi-núcleo.
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Mejorar el rendimiento de Box2D-lite}\usebeamercolor[fg]{normal text}

  \begin{block}{Objetivos}
  \begin{itemize}
    \item Demostrar y refinar conocimientos.
    \begin{itemize}
      \item Optimización y paralelización.
    \end{itemize}
    \item Permitir simular mayor cantidad de cuerpos.
    \begin{itemize}
      \item Videojuegos más detallados físicamente.
    \end{itemize}
    \item Ocupar menos tiempo CPU por fotograma.
    \begin{itemize}
      \item Lógica, IA, y gráficos duran cada vez más.
    \end{itemize}
    \item Ejecutar el motor en plataformas más modestas.
    \begin{itemize}
      \item Componentes cada vez más caros.
    \end{itemize}
  \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Primera iteración: Optimización en la broadphase}\usebeamercolor[fg]{normal text}

  \begin{block}{1º problema: Usar un único diccionario de árbitros por mundo}
    \begin{itemize}
    \item Por cada comprobación de colisión, hay que consultar y modificar el diccionario para actualizarlo.
    \item Bajo rendimiento del diccionario al haber muchas colisiones.
    \end{itemize}
  \end{block}

  \begin{exampleblock}{1ª solución: Distribuir el diccionario por cuerpo}
    \begin{itemize}
    \item Almacenar en cada uno los árbitros que lo vigilan.
    \item Mejor rendimiento al usar diccionarios más pequeños.
    \end{itemize}
  \end{exampleblock}
\end{frame}

\begin{frame}{Primera iteración: Optimización en la broadphase}\usebeamercolor[fg]{normal text}

  \begin{block}{2º problema: Comprobación de colisión de todos los
      pares de cuerpos}
    \begin{itemize}
    \item Cada comprobación es cara.
    \item Sobre todo al haber muchos cuerpos dispersos.
    \end{itemize}
  \end{block}

  \begin{exampleblock}{2ª solución: Evitar comprobar colisión con cuerpos lejanos}
  \begin{itemize}
    \item Usar una implementación de índice espacial DBVH.
    \item Por cada cuerpo, obtener los próximos.
    \item Comprobar colisión solo con ellos.
    \end{itemize}
  \end{exampleblock}
\end{frame}

\begin{frame}[fragile]{Segunda iteración: Paralelización del motor}
  \begin{columns}
    \column{0.7\textwidth}{
      \begin{block}{Paralelización de bucles con OpenMP:}
        \begin{itemize}
        \item Fácil de implementar en bucles sencillos.
        \item Sin muchos cambios en el código.
        \item Multi-plataforma.
        \end{itemize}
      \end{block}
    }
    \column{0.3\textwidth}{
      \begin{figure}
        \includegraphics[width=0.8\textwidth]{OpenMP_logo}
      \end{figure}
    }
  \end{columns}
  \begin{block}{Ejemplo}
  \begin{lstlisting}[basicstyle=\scriptsize,language=C,gobble=2]
    #pragma omp for schedule(static)
    for (int i = 0; i < (int)bodies.size(); ++i) {
      Body* b = bodies[i];
      b->velocity += dt * (gravity + b->invMass * b->force);
      b->angularVelocity += dt * b->invI * b->torque;
    }
  \end{lstlisting}
  \end{block}
\end{frame}

\begin{frame}{Segunda iteración: Paralelización del motor}
  \begin{block}{Paralelización de fases más costosas}
  \begin{itemize}
    \item Bucle de detección de colisiones
    \begin{itemize}
      \item Sin dependencias entre iteraciones gracias a la anterior iteración.
    \end{itemize}
    \item Bucles de resolución de restricciones
    \begin{itemize}
      \item Con dependencias entre iteraciones (velocidad de los cuerpos).
      \item Resuelto con lectura y actualización atómica.
    \end{itemize}
  \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Segunda iteración: Paralelización del motor}

    \begin{itemize}
    \item Bucle de aplicación de fuerza y bucle de actualización de posiciones

      \item No son muy caros.
      \item Pero sin dependencias entre iteraciones.
      \item Mayor impacto al aumentar la carga de trabajo.
    \end{itemize}

\section{Conclusiones y trabajos futuros}

\begin{frame}{Conclusiones}
  \begin{itemize}
    \item Gran complejidad de los motores de físicas
    \begin{itemize}
      \item Teórica y técnica.
    \end{itemize}
    \item La detección de colisión es la fase más importante
    \begin{itemize}
      \item Un buen índice espacial tiene un enorme impacto.
    \end{itemize}
    \item \textit{OpenMP} ofrece un buen \textit{speedup} sin apenas cambios
    \begin{itemize}
      \item Para mejorarlo aún más, se vuelve difícil.
    \end{itemize}
    \item Optimizaciones clásicas siguen marcando una gran diferencia
    \begin{itemize}
      \item \textit{Loop unrolling}, \textit{branch pruning}, \textit{float to fixed encoding} \ldots
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Trabajos futuros}
  \begin{block}{Mejorar resolución de restricciones (nueva fase más costosa)}
    \begin{itemize}
    \item Vectorizar los cálculos:
      \begin{itemize}
      \item Usar Structure of Arrays (SoA) en vez de Arrays of Structure (AoS).
      \item Aplicar el patrón de diseño \textit{Entity Component System} (ECS).
      \end{itemize}
    \item Mejorar la paralelización:
      \begin{itemize}
      \item Dividir espacio del mundo con una cuadrícula.
      \item Agrupar cuerpos por celdas.
      \item Repartir celdas entre los hilos.
      \item Menos dependecias y sincronizaciones.
      \end{itemize}
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Trabajos futuros}\usebeamercolor[fg]{normal text}
  \begin{block}{Mejorar la detección de colisiones (segunda fase más costosa)}
    \begin{itemize}
    \item \textit{Box Pruning}
      \begin{itemize}
      \item Vectorizar comprobación solapamiento geometrías.
      \item Paralelizar con Multi-SAP (híbrido cuadricula espacial y SAPs).
      \end{itemize}
    \item Mejor implementación de DBVH.
    \item Usar un índice espacial estático para los cuerpos inmóviles.
    \end{itemize}
  \end{block}
\end{frame}
